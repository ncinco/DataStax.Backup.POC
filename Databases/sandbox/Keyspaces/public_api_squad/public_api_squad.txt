CREATE KEYSPACE public_api_squad WITH replication = {'class': 'NetworkTopologyStrategy', 'us-east1': '3'}  AND durable_writes = true;

CREATE TYPE public_api_squad.phone (
    country_code int,
    number text
);

CREATE TYPE public_api_squad.address (
    street text,
    city text,
    zipcode text,
    phones map<text, phone>,
    state text
);

CREATE TABLE public_api_squad.categories (
    category_id uuid PRIMARY KEY,
    modified_date timestamp,
    name text,
    parent_category_id uuid
) WITH additional_write_policy = '99PERCENTILE'
    AND bloom_filter_fp_chance = 0.01
    AND caching = {'keys': 'ALL', 'rows_per_partition': 'NONE'}
    AND comment = ''
    AND compaction = {'class': 'org.apache.cassandra.db.compaction.UnifiedCompactionStrategy'}
    AND compression = {'chunk_length_in_kb': '64', 'class': 'org.apache.cassandra.io.compress.LZ4Compressor'}
    AND crc_check_chance = 1.0
    AND default_time_to_live = 0
    AND gc_grace_seconds = 864000
    AND max_index_interval = 2048
    AND memtable_flush_period_in_ms = 0
    AND min_index_interval = 128
    AND read_repair = 'BLOCKING'
    AND speculative_retry = '99PERCENTILE';

CREATE TABLE public_api_squad.products (
    product_id uuid PRIMARY KEY,
    category_id uuid,
    product_color text,
    product_list_price decimal,
    product_modified_date timestamp,
    product_name text,
    product_sell_end_date timestamp,
    product_sell_start_date timestamp,
    product_sku text,
    product_std_cost decimal,
    product_weight decimal,
    standard_cost decimal
) WITH additional_write_policy = '99PERCENTILE'
    AND bloom_filter_fp_chance = 0.01
    AND caching = {'keys': 'ALL', 'rows_per_partition': 'NONE'}
    AND comment = ''
    AND compaction = {'class': 'org.apache.cassandra.db.compaction.UnifiedCompactionStrategy'}
    AND compression = {'chunk_length_in_kb': '64', 'class': 'org.apache.cassandra.io.compress.LZ4Compressor'}
    AND crc_check_chance = 1.0
    AND default_time_to_live = 0
    AND gc_grace_seconds = 864000
    AND max_index_interval = 2048
    AND memtable_flush_period_in_ms = 0
    AND min_index_interval = 128
    AND read_repair = 'BLOCKING'
    AND speculative_retry = '99PERCENTILE';

CREATE TABLE public_api_squad.products_by_category (
    category_name text,
    product_sku text,
    product_name text,
    category_id uuid,
    product_color text,
    product_id uuid,
    product_list_price decimal,
    product_std_cost decimal,
    product_weight decimal,
    PRIMARY KEY (category_name, product_sku, product_name)

    AND default_time_to_live = 0
    AND gc_grace_seconds = 864000
    AND max_index_interval = 2048
    AND memtable_flush_period_in_ms = 0
    AND min_index_interval = 128
